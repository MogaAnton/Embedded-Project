#include <avr/io.h>
#include <avr/interrupt.h>

void display(char p, char c)
{
	PORTA &= 0b11110000;
	PORTC &= 0b00000000;
	switch(c){
		case 0:
		PORTC |= 0b00111111;break;
		case 1:
		PORTC |= 0b00000110;break;
		case 2:
		PORTC |= 0b01011011;break;
		case 3:
		PORTC |= 0b01001111;break;
		case 4:
		PORTC |= 0b01100110;break;
		case 5:
		PORTC |= 0b01101101;break;
		case 6:
		PORTC |= 0b01111101;break;
		case 7:
		PORTC |= 0b00000111;break;
		case 8:
		PORTC |= 0b01111111;break;
		case 9:
		PORTC |= 0b01101111;break;
	}
	if (p == 2) {
		PORTC |= 0b10000000; // Set bit 7 to 1 to enable the decimal point
	}
	switch(p){
		case 1:
		PORTA |= 0b00000001;break;
		case 2:
		PORTA |= 0b00000010;break;
		case 3:
		PORTA |= 0b00000100;break;
		case 4:
		PORTA |= 0b00001000;break;
	}
}

unsigned int ms = 0;
unsigned int s = 0;
unsigned int m = 0;
unsigned char digit = 0;
int value = 0;
int adc_l = 0, adc_h = 0, adc = 0, T = 0;
float Vin = 0, tmp = 0;

void init_ADC()
{
	ADMUX = 0b01000000;
	ADCSRA = 0b10000111;
	ADCSRA |= (1<<3);
	SREG |= (1<<7);
}

int readADC(char ch)
{
	ADMUX &= 0b11100000;
	ADMUX |= ch;
	ADCSRA |= (1<<6);
	while (ADCSRA & (1<<6));
	adc_l=ADCL;
	adc_h=ADCH;
	return ((adc_h << 8) | adc_l);
}

ISR(ADC_vect){
	adc_l = ADCL;
	adc_h = ADCH;
	adc = ((adc_h << 8) | adc_l);
	Vin = ((float)adc*5)/1024;
	tmp = Vin*1000;
	T = (int)tmp;
}

void readADC_interrupt(char ch){
	ADMUX &= 0b11100000;
	ADMUX |= ch;
	ADCSRA |= (1<<6);
}

// Func?ia de ini?ializare a timer-ului
void init_timer() {
	SREG = 1 << 7;
	TCCR0 = 0b00001011;
	TCNT0 = 0;
	OCR0 = 125;
	TIMSK |= 0b00000010;
}

// Rutina de întrerupere pentru timer
ISR(TIMER0_COMP_vect) {
	if(ms<1000)
	{
		ms++;
	}
	else
	{
		ms=0;
		readADC_interrupt(7);
	}
	digit++;  // Trecere la urm?toarea cifr? pentru afi?are
	switch(digit) {
		case 1: display(1, T % 10); break;
		case 2: display(2, (T / 10) % 10); break;
		case 3: display(3, (T / 100) % 10); break;
		case 4: display(4, T / 1000); digit = 0; break;
	}
	
}



void main()
{
	DDRA = 0b00001111; //seteaz? pinii de ie?ire
	DDRC = 0b11111111;
	init_ADC();
	init_timer();
	for(;;);
}
